import { HttpException, HttpStatus, Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Account } from './account.entity';
import { AccountDto } from './account.dto';
import { User } from '../user/user.entity';
import { Currency } from '../currency/currency.entity';
import { AccountModel } from './account.model';

/**
 * File containing method implementations of CRUD operations for an account.
 */
@Injectable()
export class AccountService {
  constructor(
    @InjectRepository(Account)
    private readonly accountRepository: Repository<Account>,
    @InjectRepository(User)
    private readonly userRepository: Repository<User>,
    @InjectRepository(Currency)
    private readonly currencyRepository: Repository<Currency>,
  ) {}

  /**
   * Method responsible for creating an account.
   */
  async createAccount(
    userId: number,
    currId: number,
    account: AccountDto,
  ): Promise<AccountModel> {
    const user: User = await this.userRepository
      .findOneOrFail(userId)
      .catch(x => {
        throw new HttpException(
          'Niepoprawny request(user)',
          HttpStatus.BAD_REQUEST,
        );
      });
    const currency: Currency = await this.currencyRepository
      .findOneOrFail(currId)
      .catch(x => {
        throw new HttpException(
          'Niepoprawny request(waluta)',
          HttpStatus.BAD_REQUEST,
        );
      });
    const newAccount: Account = Account.fromDTO(user, currency, account);
    await this.accountRepository.save(newAccount);
    return newAccount.toModel();
  }

  /**
   * Method responsible for updating an account.
   */
  async updateAccount(
    accountId: number,
    accountDto: AccountDto,
    currId?: number,
  ): Promise<AccountModel> {
    const account = await this.accountRepository
      .findOneOrFail(accountId)
      .catch(x => {
        throw new HttpException(
          'Niepoprawny request(id konta)',
          HttpStatus.BAD_REQUEST,
        );
      });
    if (currId) {
      account.currency = await this.currencyRepository
        .findOneOrFail(currId)
        .catch(x => {
          throw new HttpException(
            'Niepoprawny id waluty(currId)',
            HttpStatus.BAD_REQUEST,
          );
        });
    }
    account.budget = accountDto.budget;
    account.name = accountDto.name;
    await this.accountRepository.save(account);
    return account.toModel();
  }

  /**
   * Method responsible for deleting an account.
   */
  async deleteAccount(accountId: number) {
    const account = await this.accountRepository
      .findOneOrFail(accountId)
      .catch(x => {
        throw new HttpException(
          'Niepoprawny request(id konta)',
          HttpStatus.BAD_REQUEST,
        );
      });
    await this.accountRepository.remove(account);
  }
}
