import {
  Column,
  Entity,
  JoinColumn,
  ManyToOne,
  OneToOne,
  PrimaryGeneratedColumn,
} from 'typeorm';
import { Account } from '../account/account.entity';
import { Subcategory } from '../subcategory/subcategory.entity';

/**
 * This entity describes the Operation table.
 */
@Entity()
export class Operation {
  /**
   * Identifier of the operation.
   */
  @PrimaryGeneratedColumn()
  id: number;
  /**
   * Title of the operation.
   */
  @Column({ length: 150 })
  title: string;
  /**
   * Description of the operation.
   */
  @Column({ length: 300 })
  description: string;
  /**
   * Type of the operation (income/outcome)
   */
  @Column()
  type: boolean;
  /**
   * Value of the operation.
   */
  @Column()
  value: number;
  /**
   * Date of the operation.
   */
  @Column()
  date: Date;
  /**
   * Optional for uploading image of receipt of the operation.
   */
  @Column({ type: 'blob' })
  file: string;
  /**
   * Option that specifies if the operation is taken into the consideration.
   */
  @Column()
  intoAccount: boolean;
  /**
   * Option that specifies how many days warranty for the purchase.
   */
  @Column()
  guarantyDays: number;
  @ManyToOne(
    type => Account,
    account => account.id,
  )
  account: Account;
  @OneToOne(type => Subcategory)
  @JoinColumn()
  subcategory: Subcategory;
}
