import { Column, Entity, ManyToOne, PrimaryColumn } from 'typeorm';
import { Task } from '../task/task.entity';
import { User } from '../user/user.entity';

/**
 * This entity describes the User tasks table.
 */
@Entity()
export class Userstasks {
  /**
   * Identifier of the user task.
   */
  @PrimaryColumn()
  id: number;
  @ManyToOne(
    type => Task,
    taskId2 => taskId2.id,
  )
  taskId: Task;
  @ManyToOne(
    type => User,
    userId2 => userId2.id,
  )
  userId: User;
  /**
   * Option that specifies if the task is completed or not.
   */
  @Column()
  completed: boolean;
  /**
   * Option that specifies how many repetitions are needed to complete the task.
   */
  @Column()
  repetitionsAchieved: number;
}
