import {
  Column,
  Entity,
  ManyToOne,
  OneToMany,
  PrimaryGeneratedColumn,
} from 'typeorm';
import { Report } from '../report/report.entity';
import { Event } from '../event/event.entity';
import { Currency } from '../currency/currency.entity';
import { Cyclic } from '../cyclic/cyclic.entity';
import { User } from '../user/user.entity';
import { Operation } from '../operation/operation.entity';
import { Instalment } from '../instalment/instalment.entity';
import { AccountDto } from './account.dto';
import { AccountModel } from './account.model';

/**
 * This entity describes the Account table.
 */
@Entity()
export class Account {
  /**
   * Identifier of the account.
   */
  @PrimaryGeneratedColumn()
  id: number;
  /**
   * Name of the account.
   */
  @Column({ length: 45 })
  name: string;
  /**
   * Color of the account (front-end visualization).
   */
  @Column({ length: 45 })
  color: string;
  /**
   * Budget of the account that includes all incomes.
   */
  @Column()
  budget: number;
  /**
   * Balance of the account that includes all incomes minus operations.
   */
  @Column()
  balance: number;
  /**
   * Quantity of all account operations.
   */
  @Column()
  operationQuantity: number;
  @OneToMany(
    type => Report,
    reports => reports.id,
  )
  reports: Report[];
  @OneToMany(
    type => Event,
    events => events.id,
  )
  events: Event;
  @OneToMany(
    type => Cyclic,
    cyclics => cyclics.id,
  )
  cyclics: Cyclic[];
  @OneToMany(
    type => Operation,
    operations => operations.id,
  )
  operations: Operation[];
  @ManyToOne(
    type => Currency,
    currency => currency.id,
  )
  currency: Currency;
  @ManyToOne(
    type => User,
    user => user.id,
  )
  user: User;
  @OneToMany(
    type => Instalment,
    instalments => instalments.id,
  )
  instalments: Instalment[];

  public static fromDTO(
    user: User,
    currency: Currency,
    model: AccountDto,
  ): Account {
    const account = new Account();
    account.name = model.name;
    account.budget = model.budget;
    account.balance = model.balance;
    account.user = user;
    account.currency = currency;
    return account;
  }
  public toModel(): AccountModel {
    return new AccountModel(this);
  }
}
