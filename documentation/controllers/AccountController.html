import {
  Body,
  Controller, Delete,
  Param,
  Post,
  UseGuards,
} from '@nestjs/common';
import { AccountService } from './account.service';
import { AuthGuard } from '@nestjs/passport';
import { AccountModel } from './account.model';
import { AccountDto } from './account.dto';

/**
 * Controller responsible for CRUD operations on account.
 */
@Controller('account')
@UseGuards(AuthGuard('jwt'))
export class AccountController {
  constructor(private readonly accountService: AccountService) {}

  /**
   * This endpoint creates user's account.
   * @param userId User's identifier
   * @param currId Currency identifier
   * @param accountDto Data Transfer Object containing information about the account
   * @return Account
   */
  @Post('add/:id/:curr')
  async createAccount(
    @Param('id') userId: number,
    @Param('curr') currId: number,
    @Body() accountDto: AccountDto,
  ): Promise<AccountModel> {
    return this.accountService.createAccount(userId, currId, accountDto);
  }

  /**
   * This endpoint updates user's account.
   * @param accountId User's account identifier
   * @param accountDto Data Transfer Object containing information about the account
   * @param currId Currency identifier
   * @return Account
   */
  @Post('update/:id')
  updateAccount(
    @Param('id') accountId: number,
    @Body() accountDto: AccountDto,
    @Body('currId') currId: number,
  ): Promise<AccountModel> {
    return this.accountService.updateAccount(accountId, accountDto, currId);
  }

  /**
   * This endpoint deletes user's account.
   * @param accountId User's account identifier
   */
  @Delete('delete/:id')
  deleteAccount(@Param('id') accountId: number) {
    return this.accountService.deleteAccount(accountId);
  }
}
